const APP_NAME = "খোঁজ";

function uid() {
  return Math.random().toString(36).slice(2) + Date.now().toString(36);
}

/* ---------------------------------------------------------------------- */
/* Storage helpers — window.storage, personal scope only                  */
/* ---------------------------------------------------------------------- */

const STORAGE = {
  async getProfile() {
    try {
      const r = await window.storage.get("profile", false);
      return r ? JSON.parse(r.value) : null;
    } catch {
      return null;
    }
  },
  async setProfile(p) {
    try {
      await window.storage.set("profile", JSON.stringify(p), false);
    } catch {}
  },
  async getChatList() {
    try {
      const r = await window.storage.get("chatList", false);
      return r ? JSON.parse(r.value) : [];
    } catch {
      return [];
    }
  },
  async setChatList(list) {
    try {
      await window.storage.set("chatList", JSON.stringify(list), false);
    } catch {}
  },
  async getChat(id) {
    try {
      const r = await window.storage.get("chat:" + id, false);
      return r ? JSON.parse(r.value) : null;
    } catch {
      return null;
    }
  },
  async setChat(id, title, messages) {
    try {
      await window.storage.set(
        "chat:" + id,
        JSON.stringify({ id, title, messages, updatedAt: Date.now() }),
        false
      );
    } catch {}
  },
  async deleteAll(chatIds) {
    try {
      await window.storage.delete("profile", false);
    } catch {}
    try {
      await window.storage.delete("chatList", false);
    } catch {}
    for (const id of chatIds) {
      try {
        await window.storage.delete("chat:" + id, false);
      } catch {}
    }
  },
};

/* ---------------------------------------------------------------------- */
/* Backend API call                                                       */
/* ---------------------------------------------------------------------- */
/* The actual search + reasoning pipeline (Gemini, Groq, DuckDuckGo,      */
/* SearXNG, page fetching) lives in the separate backend project. This    */
/* frontend just calls one endpoint and renders whatever comes back.      */

// 👉 Replace with your deployed backend's URL (no trailing slash).
const API_BASE_URL = "https://your-backend-url.example.com";

async function askBackend({ question, image, language }) {
  const res = await fetch(API_BASE_URL + "/api/search", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ question, image, language }),
  });
  if (!res.ok) throw new Error("network");
  const data = await res.json();
  if (data.error) throw new Error(data.error);
  return data;
}

function GearLogo({ size = 28, state = "idle" }) {
  const spinClass = state === "thinking" ? "gear-spin-fast" : "gear-spin-slow";
  const glowClass = state === "thinking" ? "gear-glow" : "";
  const teeth = [0, 1, 2, 3, 4, 5, 6, 7];
  return (
    <div className={"gear-wrap " + glowClass} style={{ width: size, height: size }}>
      <svg viewBox="0 0 100 100" width={size} height={size} className={spinClass}>
        <defs>
          <linearGradient id="gearGrad" x1="0%" y1="0%" x2="100%" y2="100%">
            <stop offset="0%" stopColor="var(--accent)" stopOpacity="1" />
            <stop offset="100%" stopColor="var(--accent)" stopOpacity="0.55" />
          </linearGradient>
        </defs>
        <g transform="translate(50,50)">
          {teeth.map((i) => (
            <rect

          <circle r="30" fill="url(#gearGrad)" />
          <circle r="21" fill="var(--surface-solid)" />
          <circle r="5" fill="var(--accent)" />
        </g>
      </svg>
    </div>
  );
}

export default function KhojApp() {
  const [screen, setScreen] = useState("boot"); // boot | login | app
  const [theme, setTheme] = useState("light");
  const [language, setLanguage] = useState("bn");
  const [loggingIn, setLoggingIn] = useState(false);

  const [profile, setProfile] = useState(null);
  const [chats, setChats] = useState([]);
  const [currentChatId, setCurrentChatId] = useState(uid());
  const [messages, setMessages] = useState([]);

  const [composeText, setComposeText] = useState("");
  const [pendingImage, setPendingImage] = useState(null);
  const [isListening, setIsListening] = useState(false);

  const [menuOpen, setMenuOpen] = useState(false);
  const [activePanel, setActivePanel] = useState(null);
  const [sourcesFor, setSourcesFor] = useState(null);
  const [confirmingDelete, setConfirmingDelete] = useState(false);
  const [toast, setToast] = useState(null);

  const fileInputRef = useRef(null);
  const recognitionRef = useRef(null);
  const toastTimerRef = useRef(null);
  const scrollRef = useRef(null);

  const t = useCallback((key) => STRINGS[language][key], [language]);

  /* boot sequence */
  useEffect(() => {
    let timer;
    (async () => {
      const existing = await STORAGE.getProfile();
      timer = setTimeout(async () => {
        if (existing) {
          setProfile(existing);
          const list = await STORAGE.getChatList();
          setChats(list);
          setScreen("app");
        } else {
          setScreen("login");
        }
      }, 2200);
    })();
    return () => clearTimeout(timer);
  }, []);

  useEffect(() => {
    if (scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
    }
  }, [messages]);

  function showToast(msg) {
    setToast(msg);
    if (toastTimerRef.current) clearTimeout(toastTimerRef.current);
    toastTimerRef.current = setTimeout(() => setToast(null), 2600);
  }

  async function handleGoogleLogin() {
    setLoggingIn(true);
    let p = await STORAGE.getProfile();
    if (!p) {
      p = {
        name: language === "en" ? "You" : "আপনি",
        email: "user@gmail.com",
        joinDate: new Date().toISOString(),
        totalQuestions: 0,
        totalChats: 0,
      };
      await STORAGE.setProfile(p);
    }
    const list = await STORAGE.getChatList();
    setChats(list);
    setProfile(p);
    setLoggingIn(false);
    setScreen("app");
  }

  function handleLogout() {
    setMenuOpen(false);
    setActivePanel(null);
    setMessages([]);
    setCurrentChatId(uid());
    setScreen("login");
  }

  async function handleDeleteAllData() {
    const ids = chats.map((c) => c.id);
    await STORAGE.deleteAll(ids);
    setProfile(null);
    setChats([]);
    setMessages([]);
    setCurrentChatId(uid());
    setConfirmingDelete(false);
    setActivePanel(null);
    setMenuOpen(false);
    setScreen("login");
  }

  function handleNewChat() {
    if (messages.length === 0) {
      setMenuOpen(false);
      return;
    }
    setCurrentChatId(uid());
    setMessages([]);
    setPendingImage(null);
    setComposeText("");
    setMenuOpen(false);
  }

  async function handleOpenChat(id) {
    const chat = await STORAGE.getChat(id);
    if (chat) {
      setCurrentChatId(id);
      setMessages(chat.messages || []);
    }
    setActivePanel(null);
  }

  async function persistTurn(finalMessages, fallbackTitle) {
    let list = chats;
    let entry = list.find((c) => c.id === currentChatId);
    const title = (entry && entry.title) || fallbackTitle || t("newChat");
    if (!entry) {
      entry = { id: currentChatId, title, updatedAt: Date.now() };
      list = [entry, ...list];
    } else {
      entry = { ...entry, updatedAt: Date.now() };
      list = list.map((c) => (c.id === currentChatId ? entry : c));
    }
    setChats(list);
    await STORAGE.setChatList(list);
    await STORAGE.setChat(currentChatId, title, finalMessages);

    const updatedProfile = {
      ...profile,
      totalQuestions: ((profile && profile.totalQuestions) || 0) + 1,
      totalChats: list.length,
    };
    setProfile(updatedProfile);
    await STORAGE.setProfile(updatedProfile);
  }

  async function runTurn(userMsg, aiMsgId, baseMessages) {
    let result, status;
    try {
      result = await askBackend({
        question: userMsg.text,
        image: pendingImageRefToPayload(userMsg.imagePayload),
        language,
      });
      status = "done";
    } catch (e) {
      result = null;
      status = "error";
    }
    const finalMessages = baseMessages.map((m) =>
      m.id === aiMsgId ? { ...m, status, content: result } : m
    );
    setMessages(finalMessages);
    if (status === "done") {
      const fallbackTitle =
        userMsg.text || (userMsg.image ? t("imageSearchTitle") : t("newChat"));
      await persistTurn(finalMessages, fallbackTitle);
    }
  }

  function pendingImageRefToPayload(imagePayload) {
    return imagePayload || null;
  }

  async function handleSend() {
    const text = composeText.trim();
    if (!text && !pendingImage) return;
    const userMsg = {
      id: uid(),
      role: "user",
      text,
      image: pendingImage ? pendingImage.dataUrl : null,
      imagePayload: pendingImage
        ? { data: pendingImage.base64, mediaType: pendingImage.mediaType }
        : null,
    };
    const aiMsg = { id: uid(), role: "ai", status: "thinking", content: null };
    const baseMessages = [...messages, userMsg, aiMsg];
    setMessages(baseMessages);
    setComposeText("");
    setPendingImage(null);
    await runTurn(userMsg, aiMsg.id, baseMessages);
  }

  async function handleRetry(aiMsgId) {
    const idx = messages.findIndex((m) => m.id === aiMsgId);
    if (idx < 1) return;
    const userMsg = messages[idx - 1];
    const baseMessages = messages.map((m) =>
      m.id === aiMsgId ? { ...m, status: "thinking", content: null } : m
    );
    setMessages(baseMessages);
    await runTurn(userMsg, aiMsgId, baseMessages);
  }

  function handleFilePicked(e) {
    const file = e.target.files && e.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = () => {
      const dataUrl = String(reader.result);
      const commaIdx = dataUrl.indexOf(",");
      const meta = dataUrl.slice(0, commaIdx);
      const base64 = dataUrl.slice(commaIdx + 1);
      const match = meta.match(/data:(.*);base64/);
      const mediaType = match ? match[1] : "image/jpeg";
      setPendingImage({ dataUrl, base64, mediaType });
    };
    reader.readAsDataURL(file);
    e.target.value = "";
  }

  function toggleListen() {
    const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
    if (!SR) {
      showToast(t("voiceUnsupported"));
      return;
    }
    if (isListening) {
      if (recognitionRef.current) recognitionRef.current.stop();
      return;
    }
    const rec = new SR();
    rec.lang = language === "en" ? "en-US" : "bn-BD";
    rec.interimResults = false;
    rec.maxAlternatives = 1;
    rec.onresult = (e) => {
      const transcript = e.results[0][0].transcript;
      setComposeText((prev) => (prev ? prev + " " : "") + transcript);
    };
    rec.onerror = () => {
      setIsListening(false);
      showToast(t("voiceError"));
    };
    rec.onend = () => setIsListening(false);
    recognitionRef.current = rec;
    rec.start();
    setIsListening(true);
  }

  const isBusy = messages.some((m) => m.status === "thinking");
  const sourcesMessage = sourcesFor ? messages.find((m) => m.id === sourcesFor) : null;
  const sourcesList =
    sourcesMessage && sourcesMessage.content && Array.isArray(sourcesMessage.content.sources)
      ? sourcesMessage.content.sources
      : [];

  /* ---------------------------- render: boot ---------------------------- */

  if (screen === "boot") {
    return (
      <Shell theme={theme}>
        <div className="splash">
          <GearLogo size={84} state="idle" />
          <div className="splash-name">{APP_NAME}</div>
        </div>
      </Shell>
    );
  }

  /* ---------------------------- render: login ---------------------------- */

  if (screen === "login") {
    return (
      <Shell theme={theme}>
        <div className="login">
          <div className="login-top">
            <GearLogo size={64} state="idle" />
            <div className="login-name">{APP_NAME}</div>
            <div className="login-tagline">{t("tagline")}</div>
          </div>
          <div className="login-bottom">
            <button className="google-btn" onClick={handleGoogleLogin} disabled={loggingIn}>
              <GoogleMark />
              <span>{loggingIn ? t("signingIn") : t("signIn")}</span>
            </button>
            <div className="login-mock-notice">{t("mockNotice")}</div>
          </div>
        </div>
      </Shell>
    );
  }

  /* ---------------------------- render: app ---------------------------- */

  const aiState = isBusy ? "thinking" : "idle";

  return (
    <Shell theme={theme}>
      <div className="app-root">
        <div className="top-bar">
          <button className="icon-btn" onClick={() => setMenuOpen(true)} aria-label="menu">
            <Menu size={22} />
          </button>
          <GearLogo size={26} state={aiState} />
          <button className="icon-btn" onClick={handleNewChat} aria-label="new chat">
            <Plus size={22} />
          </button>
        </div>

        {messages.length === 0 ? (
          <div className="greeting">{t("greeting")}</div>
        ) : (
          <div className="chat-scroll" ref={scrollRef}>
            {messages.map((m) =>
              m.role === "user" ? (
                <UserBubble key={m.id} msg={m} />
              ) : (
                <AiCard
                  key={m.id}
                  msg={m}
                  t={t}
                  onOpenSources={() => setSourcesFor(m.id)}
                  onRetry={() => handleRetry(m.id)}
                />
              )
            )}
          </div>
        )}

        <ComposeBar
          t={t}
          value={composeText}
          onChange={setComposeText}
          onSend={handleSend}
          disabled={isBusy}
          isListening={isListening}

function AiCard({ msg, t, onOpenSources, onRetry }) {
  if (msg.status === "thinking") {
    return (
      <div className="msg-row">
        <div className="thinking-row">
          <GearLogo size={20} state="thinking" />
          <span className="thinking-text">{t("thinking2")}</span>
        </div>
      </div>
    );
  }
  if (msg.status === "error") {
    return (
      <div className="msg-row">
        <div className="ai-card ai-card-error">
          <div>{t("errorAnswer")}</div>
          <button className="retry-btn" onClick={onRetry}>
            {t("retry")}
          </button>
        </div>
      </div>
    );
  }
  const content = msg.content || {};
  const hasSources = Array.isArray(content.sources) && content.sources.length > 0;
  return (
    <div className="msg-row">
      <div className="ai-card">
        {hasSources && (
          <div className="ai-card-header">
            <button
              className="globe-btn"
              onClick={onOpenSources}
              aria-label={t("sourcesSheetTitle") + " (" + content.sources.length + ")"}
            >
              <Globe size={16} />
            </button>
          </div>
        )}
        {content.shortAnswer && <div className="ai-short">{content.shortAnswer}</div>}
        {content.detailedAnswer && <div className="ai-detailed">{content.detailedAnswer}</div>}
        {Array.isArray(content.table && content.table.headers) && content.table.rows && (
          <div className="table-wrap">
            <table>
              <thead>
                <tr>
                  {content.table.headers.map((h, i) => (
                    <th key={i}>{h}</th>
                  ))}
                </tr>
              </thead>
              <tbody>
                {content.table.rows.map((row, ri) => (
                  <tr key={ri}>
                    {row.map((cell, ci) => (
                      <td key={ci}>{cell}</td>
                    ))}
                  </tr>
                ))}
              </tbody>
            </table>
          </div>
        )}
        {Array.isArray(content.bullets) && content.bullets.length > 0 && (
          <ul className="bullet-list">
            {content.bullets.map((b, i) => (
              <li key={i}>{b}</li>
            ))}
          </ul>
        )}
        {content.example && (
          <div className="example-box">
            <span className="example-label">{t("exampleLabel")}</span>
            {content.example}
          </div>
        )}
      </div>
    </div>
  );
}

function SourceLogo({ logo, domain }) {
  const [failed, setFailed] = useState(false);
  if (!logo || failed) {
    return <div className="source-logo-fallback">{((domain || "?")[0] || "?").toUpperCase()}</div>;
  }
  return <img className="source-logo" src={logo} alt="" onError={() => setFailed(true)} />;
}

  );
}
